\newcount\r
\newcount\g
\newcount\b
\r=11
\loop
{
    \g=11
    \b=11
    \loop
        \pdfliteral{.\the\r}\pdfliteral{.\the\g}\pdfliteral{.\the\b}\pdfliteral{rg} X
    \advance \g by 5
    \advance \b by 3
    \unless\ifnum \g>99
    \repeat
    \endgraf
}
\advance \r by 5
\unless\ifnum \r>99
\repeat

\end
